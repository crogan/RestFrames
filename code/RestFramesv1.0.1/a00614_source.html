57 
58  TVector3 v_inv_P = inv_P.Vect();
59  v_inv_P = v_inv_P - v_inv_P.Dot(m_Axis.Unit())*m_Axis.Unit();
60  inv_P.SetVectM(v_inv_P, inv_P.M());
61
