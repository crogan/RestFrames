57  void VisibleState::AddFrame(const RestFrame& frame){
58  if(IsEmpty()) return;
59 
60  if(!frame) return;
61  if(!frame.IsVisibleFrame() || !frame.IsRecoFrame()) return;
62  m_FramePtr = static_cast<const VisibleRecoFrame*>(&frame);
63  m_Frames.Clear();
64  m_Frames += frame;
65  }
