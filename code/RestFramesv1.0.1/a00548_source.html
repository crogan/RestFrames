51  void GaussianPDF::Optimize(const RFDataVar& w, double beta){
52  int Ndata = GetNData();
53 
54  m_N = 0;
55  m_mean = 0;
56  m_mean2 = 0;
57  for(int i = 0; i < Ndata; i++){
58  m_N += w[i];
59  m_mean += w[i]*GetVar()[i];
60  m_mean2 += w[i]*GetVar()[i]*GetVar()[i];
61  }
62  m_mean /= m_N;
63  m_mean2 /= m_N;
64 
65  // posterior parameters
66  m_gaus_beta = m_gaus_beta0 + m_N;
67  m_gaus_m = (m_gaus_beta0*m_gaus_m0 + m_N*m_mean)/m_gaus_beta;
68  m_gamma_a = m_gamma_a0 + m_N/2.;
69  m_gamma_b = m_gamma_b0 +
70  m_N*((m_mean2 - m_mean*m_mean) +
71  m_gaus_beta0*(m_mean - m_gaus_m0)*(m_mean - m_gaus_m0)/m_gaus_beta)/2.;
72 
73  // predictive parameters
74  m_pred_lambda = m_gaus_beta/(m_gaus_beta + 1.)*m_gamma_a/m_gamma_b;
75  m_pred_norm = sqrt(acos(-1.))/TMath::Beta(m_gamma_a,0.5)*
76  pow(m_pred_lambda/acos(-1.)/m_gamma_a/2., 0.5);
77  m_pred_var = m_gamma_a/m_pred_lambda/(m_gamma_a - 1.);
78 
79  // pre-calc parameters
80  m_pre_norm = 0.5*(-log(2.*acos(-1.)) - 1./m_gaus_beta +
81  Digamma(m_gamma_a) - log(m_gamma_b));
82 
83  double LogL = 0.;
84 
85  LogL = log(m_gaus_beta0/m_gaus_beta) + 1. - m_gaus_beta0/m_gaus_beta
86  -m_gaus_beta0*m_gamma_a/m_gamma_b*(m_gaus_m-m_gaus_m0)*(m_gaus_m-m_gaus_m0);
87 
88  LogL = LogL/2. + TMath::LnGamma(m_gamma_a) - TMath::LnGamma(m_gamma_a0)
89  + m_gamma_a0*log(m_gamma_b0) - m_gamma_a*log(m_gamma_b)
90  + m_gamma_a*(1.-m_gamma_b0/m_gamma_b)
91  + (m_gamma_a0-m_gamma_a)*(Digamma(m_gamma_a) - log(m_gamma_b));
92 
93  LogL += m_N*(m_pre_norm - 0.5*m_gamma_a/m_gamma_b*
94  (m_mean2-m_mean*m_mean + (m_mean-m_gaus_m)*(m_mean-m_gaus_m)));
95  SetLogL(LogL);
96  }
