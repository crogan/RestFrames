154  void RSelfAssemblingFrame::AssembleRecursive(RestFrame* framePtr, vector<RestFrame*>& frames, vector<TLorentzVector>& Ps){
155  int Ninput = frames.size();
156  if(Ninput <= 1){
157  for(int i = 0; i < Ninput; i++) framePtr->AddChildFrame(frames[i]);
158  return;
159  }
160 
161  TLorentzVector TOT(0.,0.,0.,0.);
162  for(int i = 0; i < Ninput; i++) TOT += Ps[i];
163  TVector3 boost = TOT.BoostVector();
164  for(int i = 0; i < Ninput; i++){
165  Ps[i].Boost(-boost);
166  }
167 
168  int ip_max[2];
169  int jp_max[2];
170  for(int i = 0; i < 2; i++) ip_max[i] = -1;
171  for(int i = 0; i < 2; i++) jp_max[i] = -1;
172  double val_max = -1.;
173  // Loop over all 2-jet seed probes
174  int ip[2], jp[2];
175  for(ip[0] = 0; ip[0] < Ninput-1; ip[0]++){
176  for(ip[1] = ip[0]+1; ip[1] < Ninput; ip[1]++){
177  TVector3 nRef = Ps[ip[0]].Vect().Cross(Ps[ip[1]].Vect());
178  int Nhem[2];
179  TLorentzVector hem[2];
180  for(int i = 0; i < 2; i++){
181  Nhem[i] = 0;
182  hem[i].SetPxPyPzE(0.,0.,0.,0.);
183  }
184  // Loop over all jets
185  for(int i = 0; i < Ninput; i++){
186  if((i == ip[0]) || (i ==ip[1])) continue;
187  int ihem = int(Ps[i].Vect().Dot(nRef) > 0.);
188  Nhem[ihem]++;
189  hem[ihem] += Ps[i];
190  }
191  // assign 2 probes
192  for(jp[0] = 0; jp[0] < 2; jp[0]++){
193  for(jp[1] = 0; jp[1] < 2; jp[1]++){
194  if(jp[0] == jp[1] && Nhem[(jp[0]+1)%2] == 0) continue;
195  TLorentzVector hem_probes[2];
196  for(int i = 0; i < 2; i++) hem_probes[i] = hem[i];
197  for(int i = 0; i < 2; i++) hem_probes[jp[i]] += Ps[ip[i]];
198  double val = hem_probes[0].P() + hem_probes[1].P();
199  if(val > val_max){
200  val_max = val;
201  for(int i = 0; i < 2; i++) ip_max[i] = ip[i];
202  for(int i = 0; i < 2; i++) jp_max[i] = jp[i];
203  }
204  }
205  }
206  }
207  }

210  vector<TLorentzVector> child_Ps[2];
211  TLorentzVector hem[2];
212  for(int i = 0; i < 2; i++){
213  hem[i].SetPxPyPzE(0.,0.,0.,0.);
214  }
215  for(int i = 0; i < 2; i++){
216  child_frames[jp_max[i]].push_back(frames[ip_max[i]]);
217  child_Ps[jp_max[i]].push_back(Ps[ip_max[i]]);
218  hem[jp_max[i]] += Ps[ip_max[i]];
219  }
220  TVector3 nRef = Ps[ip_max[0]].Vect().Cross(Ps[ip_max[1]].Vect());
221  for(int i = 0; i < Ninput; i++){
222  if((i == ip_max[0]) || (i == ip_max[1])) continue;
223  int ihem = int(Ps[i].Vect().Dot(nRef) > 0.);
224  child_frames[ihem].push_back(frames[i]);
225  child_Ps[ihem].push_back(Ps[i]);
226  hem[ihem] += Ps[i];
227  }
228 
229  int flip = int(hem[1].M() > hem[0].M());
230  for(int i = 0; i < 2; i++){
231  int j = (i+flip)%2;
232  if(child_frames[j].size() == 1){
233  framePtr->AddChildFrame(child_frames[j][0]);
234  } else {
235  RestFrame* new_framePtr = GetNewDecayFrame(GetName(),GetTitle());

238  }
